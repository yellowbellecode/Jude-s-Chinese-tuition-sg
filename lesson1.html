    <script>

        // Vocabulary list

        const vocabWords = [

            { chinese: "爸爸", pinyin: "bà ba", english: "father" },

            { chinese: "妈妈", pinyin: "mā ma", english: "mother" },

            { chinese: "儿子", pinyin: "ér zi", english: "son" },

            { chinese: "女儿", pinyin: "nǚ ér", english: "daughter" }

        ];

        

        let currentWordIndex = 0;

        

        // Audio playback using Web Speech API

        function playAudio(text) {

            const utterance = new SpeechSynthesisUtterance(text);

            utterance.lang = 'zh-CN'; // Chinese

            utterance.rate = 0.8; // Slower for learning

            window.speechSynthesis.speak(utterance);

        }

        

        // Play the current practice word

        function playPracticeWord() {

            const word = vocabWords[currentWordIndex].chinese;

            playAudio(word);

        }

        

        // Check if the written answer is correct

        function checkWriting() {

            const input = document.getElementById('writingInput').value.trim();

            const correct = vocabWords[currentWordIndex].chinese;

            const feedback = document.getElementById('feedback');

            

            if (input === correct) {

                feedback.className = 'feedback correct';

                feedback.innerHTML = '✓ Correct! Well done!';

                

                // Move to next word after correct answer

                currentWordIndex = (currentWordIndex + 1) % vocabWords.length;

                document.getElementById('currentPracticeWord').textContent = 

                    vocabWords[currentWordIndex].chinese;

                document.getElementById('writingInput').value = '';

            } else {

                feedback.className = 'feedback incorrect';

                feedback.innerHTML = `✗ Try again. The correct answer is: ${correct}`;

            }

        }

        

        // Optional: Add pinyin input method

        document.getElementById('writingInput').addEventListener('keypress', function(e) {

            if (e.key === 'Enter') {

                checkWriting();

            }

        });

    </script>
